# Standard minimum interval (in seconds) it takes to regrow fruits
min_regrow_interval (Minimum Regrow Interval) int 300

# Standard maximum interval (in seconds) it takes to regrow fruits
max_regrow_interval (Maximum Regrow Interval) int 1500

# Chance to stop regrowth of a fruit (Causes trees to bear less fruit over time; set to 0 to disable)
regrowth_stop_chance (Regrowth Stop Chance) float 0.02
